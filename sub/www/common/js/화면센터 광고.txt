var login_myURL:String; 

login1.addEventListener(MouseEvent.CLICK , login);
function login(e:MouseEvent):void{
      login_myURL="javascript:openWindow('팝업창 주소','팝업창 이름', 너비 , 높이 )";
      navigateToURL(new URLRequest(login_myURL) , "_self"); 
	
}
